31 json_value* generate_json_array_of_components(LH_List *components) {
32  if (components == NULL)
33  return json_array_new(0);
34 
35  int num_status = components->size;
36  json_value *components_json_array = json_array_new(num_status);
37 
38  int j;
39  for (j = 0; j < num_status; j++) {
40  LH_Component *component = (LH_Component*) lh_list_get(components, j);
41  json_value *json_component = json_object_new(2);
42  json_value *name_value = json_string_new(component->name);
43  json_object_push(json_component, "name", name_value);
44 
45  json_value *type_value;
46  switch (component->type) {
47  case LH_TYPE_INTEGER:
48  type_value = json_string_new("integer");
49  break;
50  case LH_TYPE_FLOAT:
51  type_value = json_string_new("float");
52  break;
53  case LH_TYPE_STRING:
54  type_value = json_string_new("string");
55  break;
56  case LH_TYPE_TIMESTAMP:
57  type_value = json_string_new("timestamp");
58  break;
59  case LH_TYPE_BOOLEAN:
60  type_value = json_string_new("boolean");
61  break;
62  default:
63  log_warn("Unable to generate descriptor: unrecognized type of component. Skipping.");
64  continue;
65  }
66 
67  json_object_push(json_component, "type", type_value);
68  json_array_push(components_json_array, json_component);
69  }
70  return components_json_array;
71 }

137

143 
144  // add events
145  if (device->events != NULL)
146  add_events_to_descriptor(device, descriptor);
147  else
148  json_object_push(descriptor, "eventList", json_array_new(0));
149 
150  // add actions
151  if (device->actions != NULL)
152  add_actions_to_descriptor(device, descriptor);
153  else
154  json_object_push(descriptor, "actionList", json_array_new(0));
155 
156  char *json_descriptor = malloc(json_measure(descriptor) * sizeof *json_descriptor);
157  json_serialize(json_descriptor, descriptor);
158  json_builder_free(descriptor);
159  return json_descriptor;

255 LH_Dict* process_arguments_attribute(StunAttribute *attr, LH_Action *action) {
256  int num_args = attr->bytes[0];
257  if (num_args > 8 || num_args < 0) {
258  log_error("Maximum 8 arguments are allowed.");
259  return NULL;
260  }
261 
262  uint8_t string_mask = attr->bytes[num_args + 1];
263  LH_Dict *processed_args = lh_dict_new();
264  int j = 0;
265  int position = num_args + 2;
266  for (j = 0; j < num_args; j++) {
267  int arg_len = attr->bytes[j + 1];
268  char *arg_name;
269  void *arg_value;
270  if (is_string_arg(string_mask, j)) {
271  int arg_value_len = byte_array_to_uint16_t(attr->bytes + position);
272  int arg_name_len = byte_array_to_uint16_t(attr->bytes + position + 2);
273  arg_name = malloc((arg_name_len + 1) * sizeof *arg_name);
274  arg_value = malloc((arg_value_len + 1) * sizeof (char));
275  memcpy(arg_value, attr->bytes + position + 4, arg_value_len);
276  memcpy(arg_name, attr->bytes + position + arg_value_len + 4, arg_name_len);
277  ((char*) arg_value)[arg_value_len] = 0;
278  arg_name[arg_name_len] = 0;
279  } else {
280  arg_name = malloc((arg_len + 1) * sizeof *arg_name);
281  memcpy(arg_name, attr->bytes + position + 4, arg_len);
282  arg_name[arg_len] = 0;
283  LH_ComponentType type = action_get_component_type(arg_name, action);
284  arg_value = malloc(sizeof (uint32_t));
285  switch (type) {
286  case LH_TYPE_INTEGER:
287  case LH_TYPE_BOOLEAN:
288  case LH_TYPE_TIMESTAMP:
289  *((uint32_t *) arg_value) = byte_array_to_uint32(attr->bytes + position);
290  break;
291  case LH_TYPE_FLOAT:
292  *((float *) arg_value) = byte_array_to_float(attr->bytes + position);
293  break;
294  case LH_TYPE_STRING:
295  case LH_TYPE_NO_TYPE:
296  log_error("Arguments attribute: expected four byte argument type but received string or no type.");
297  free_args_dictionary(processed_args);
298  return NULL;
299  }
300  }
301  lh_dict_put(processed_args, arg_name, arg_value);
302  free(arg_name);
303  position = position + arg_len + 4;
304  }
305  return processed_args;
306 }

308 void perform_action(StunMessage *message) {
309  StunAttribute attr_name;
310  int attr_present = stun_get_attribute(message, ATTR_NAME, &attr_name);
311  if (!attr_present)
312  return;
313  char* action_name = malloc((attr_name.length + 1) * sizeof *action_name);
314  memcpy(action_name, attr_name.bytes, attr_name.length);
315  action_name[attr_name.length] = 0;
316  int num_actions_of_device = this_device.actions->size;
317  int j;
318  LH_Action *action_to_execute = NULL;
319  for (j = 0; j < num_actions_of_device; j++) {
320  LH_Action *action = (LH_Action *) lh_list_get(this_device.actions, j);
321  if (strcmp(action_name, action->name) == 0) {
322  action_to_execute = action;
323  break;
324  }
325  }
326  if (action_to_execute == NULL) {
327  char *log_msg = lh_get_message_str("Device has no action with name %s", action_name);
328  log_warn(log_msg);
329  free(log_msg);
330  free(action_name);
331  return;
332  }
333 
334  StunAttribute attr_arguments;
335  attr_present = stun_get_attribute(message, ATTR_ARGUMENTS, &attr_arguments);
336  if (!attr_present) {
337  free(action_name);
338  return;
339  }
340 
341  LH_Dict *arguments = process_arguments_attribute(&attr_arguments, action_to_execute);
342  if (arguments == NULL) {
343  log_error("Could not process arguments attribute. Action not performed.");
344  free(action_name);
345  return;
346  }
347 
348  action_to_execute->action_function(arguments);
349  free_args_dictionary(arguments);
350  free(action_name);
351 }

353 uint8_t* build_arguments_attribute(LH_Device *device, int *len) {
354  int num_status = device->status_components->size;
355  if (num_status > 8) {
356  log_error("More than 8 components are not allowed.");
357  return NULL;
358  }
359  int size_to_alloc = 2 + num_status;
360  int j;
361  for (j = 0; j < num_status; j++) {
362  LH_Component *status = lh_list_get(device->status_components, j);
363  int name_len = strlen(status->name);
364  int value_len = 0;
365  if (status->type == LH_TYPE_STRING) {
366  char *value = (char *) status->value;
367  value_len = strlen(value);
368  }
369  size_to_alloc += name_len + value_len + 4;
370  }
371  uint8_t *attr_arguments_bytes = malloc(size_to_alloc * sizeof *attr_arguments_bytes);
372  memset(attr_arguments_bytes, size_to_alloc, 0);
373  *len = size_to_alloc;
374  attr_arguments_bytes[0] = num_status;
375  int position = 2 + num_status;
376  for (j = 0; j < num_status; j++) {
377  LH_Component *status = lh_list_get(device->status_components, j);
378  int str_len = 0;
379  int name_len = 0;
380  switch (status->type) {
381  case LH_TYPE_INTEGER:
382  case LH_TYPE_TIMESTAMP:
383  uint32_to_byte_array(*(uint32_t *) status->value, attr_arguments_bytes + position);
384  position += 4;
385  str_len = strlen((char*) status->name);
386  memcpy(attr_arguments_bytes + position, status->name, str_len);
387  // update status length
388  *(attr_arguments_bytes + j + 1) = (uint8_t) str_len;
389  position += str_len;
390  break;
391  case LH_TYPE_BOOLEAN:
392  if (*(int *) status->value)
393  uint32_to_byte_array((uint32_t) 1, attr_arguments_bytes + position);
394  else
395  uint32_to_byte_array((uint32_t) 0, attr_arguments_bytes + position);
396  position += 4;
397  str_len = strlen((char*) status->name);
398  memcpy(attr_arguments_bytes + position, status->name, str_len);
399  position += str_len;
400  *(attr_arguments_bytes + j + 1) = (uint8_t) str_len;
401  break;
402  case LH_TYPE_FLOAT:
403  float_to_byte_array(*(float *) status->value, attr_arguments_bytes + position);
404  position += 4;
405  str_len = strlen((char*) status->name);
406  memcpy(attr_arguments_bytes + position, status->name, str_len);
407  position += str_len;
408  *(attr_arguments_bytes + j + 1) = (uint8_t) str_len;
409  break;
410  case LH_TYPE_STRING:
411  // set string mask bit
412  *(attr_arguments_bytes + num_status + 1) = *(attr_arguments_bytes + num_status + 1) | (1 << j);
413  str_len = strlen((char*) status->value);
414  uint16_to_byte_array((uint16_t) str_len, attr_arguments_bytes + position);
415  name_len = strlen(status->name);
416  uint16_to_byte_array((uint16_t) name_len, attr_arguments_bytes + position + 2);
417  position += 4;
418  memcpy(attr_arguments_bytes + position, status->value, str_len);
419  position += str_len;
420  memcpy(attr_arguments_bytes + position, status->name, name_len);
421  position += str_len;
422  *(attr_arguments_bytes + j + 1) = (uint8_t) str_len + name_len;
423  break;
424  case LH_TYPE_NO_TYPE:
425  default:
426  break;
427  }
428 
429  }
430  return attr_arguments_bytes;
431 }

459 void process_messages() {
460  uint16_t length;
461  uint8_t *bytes = lh_receive_from_server(&length);
462  if (bytes == NULL)
463  return;
464 
465  StunMessage message;
466  if (!stun_process_stun_message(bytes, length, &message)) {
467  log_warn("Discarding malformed message received.");
468  free(bytes);
469  return;
470  }
471 
472  // TODO add here check of trId to avoid processing duplicated messages
473 
474  if (!stun_is_integrity_correct(&message, this_device.api_key)) {
475  log_warn("Discarding message received with bad integrity.");
476  free(bytes);
477  return;
478  }
479 
480  uint16_t method, class;
481  stun_get_method_and_class(&message, &method, &class);
482  char str[100];
483  encode_hex(message.bytes + 8, 12, str);
484  printf("STUN message received: class %0x, method %0x, trId %s\n", class, method, str);
485  if (class == CL_ERROR) {
486  // check for bad timestamp message
487  StunAttribute attribute;
488  if (stun_get_attribute(&message, ATTR_ERROR_CODE, &attribute)) {
489  int hundreds = (int) attribute.bytes[2];
490  int remainder = (int) attribute.bytes[3];
491  int error_code = 100 * hundreds + remainder;
492  if (error_code == ERR_BAD_TIMESTAMP)
493  log_warn("Bad timestamp message received.");
494  }
495  if (stun_get_attribute(&message, ATTR_SERVER_TIME, &attribute)) {
496  uint32_t server_time = byte_array_to_uint32(attribute.bytes);
497  lh_update_time_offset(server_time);
498  }
499  free(bytes);
500  return;
501  }
502 
503  if (class == CL_REQUEST) {
504  switch (method) {
505  case M_ACTION:
506  perform_action(&message);
507  send_success_response(&message);
508  break;
509  case M_STATUS_REQUEST:
510  send_status(&message);
511  break;
512  default:
513  break;
514  }
515  }
516  free(bytes);
517  return;
518 }

520 int lh_start_device(LH_Device *device, char *device_name, char *username, char *password) {
521  device->name = device_name;
522  device->username = username;
523  char *apikey = lh_api_get_api_key(username, password);
524  if (apikey == NULL)
525  return 0;
526 
527  // initialize device
528  device->actions = NULL;
529  device->events = NULL;
530  device->status_components = NULL;
531  device->api_key = apikey;
532 
533  char* uuid = lh_storage_get_uuid(device->name);
534  if (uuid == NULL) {
535  // device has not been registered before, do it now
536  if (!lh_api_register_device(device)) {
537  log_error("Device registration failed.");
538  free(apikey);
539  return 0;
540  } else {
541  lh_storage_save_uuid(device->name, device->uuid);
542  }
543  } else {
544  device->uuid = uuid;
545  }
546  // call user defined setup function
547  log_info("Configuring device")
548  setup();
549 
550  // send descriptor file
551  char *device_descriptor = generate_descriptor(device);
552  if (LOG_DESCRIPTOR)
553  log_info(device_descriptor);
554  int success;
555  do {
556  success = retry_send_device_descriptor(device, device_descriptor);
557  } while (!success);
558  free(device_descriptor);
559 
560  // start session in Lhings
561  do {
562  success = retry_start_session(device);
563  } while (!success);
564  log_info("Session started!")
565  // send first keepalive
566  send_keepalive(device);
567  // main loop
568  while (1) {
569  send_keepalive_if_needed(device);
570  process_messages();
571  loop_driver();
572  }
573 }

641 char *build_structured_payload(LH_List *components) {
642  int j;
643  json_value *json = json_array_new(components->size);
644  for (j = 0; j < components->size; j++) {
645  json_value *json_component = json_object_new(2);
646  LH_Component *component = lh_list_get(components, j);
647  json_object_push(json_component, "name", json_string_new(component->name));
648  switch (component->type) {
649  case LH_TYPE_BOOLEAN:
650  json_object_push(json_component, "value", json_boolean_new((int) *(uint32_t *) component->value));
651  break;
652  case LH_TYPE_INTEGER:
653  case LH_TYPE_TIMESTAMP:
654  json_object_push(json_component, "value", json_integer_new((int) *(uint32_t *) component->value));
655  break;
656  case LH_TYPE_FLOAT:
657  json_object_push(json_component, "value", json_double_new((double) *(float *) component->value));
658  break;
659  case LH_TYPE_STRING:
660  json_object_push(json_component, "value", json_string_new((char *) component->value));
661  break;
662  case LH_TYPE_NO_TYPE:
663  default:
664  break;
665  }
666  json_array_push(json, json_component);
667  }
668  char *json_payload = malloc (json_measure(json) * sizeof *json_payload);
669  json_serialize(json_payload, json);
670  json_builder_free(json);
671  return json_payload;
672 }
