38 size_t writefunc(void *ptr, size_t size, size_t nmemb, struct string *s) {
39  size_t new_len = s->len + size*nmemb;
40  s->ptr = realloc(s->ptr, new_len + 1);
41  if (s->ptr == NULL) {
42  log_error("Could not allocate memory (realloc) while reading response from server\n");
43  return 0;
44  }
45  memcpy(s->ptr + s->len, ptr, size * nmemb);
46  s->ptr[new_len] = '\0';
47  s->len = new_len;
48 
49  return size*nmemb;
50 }

59 LH_HttpResponse* http_execute_request(const char* url, LH_Dict* headers, uint8_t method, const char* request_body) {
60  CURL *curl = curl_easy_init();
61  if (curl) {
62  CURLcode res;
63  LH_List* keys_of_headers = lh_dict_get_keys(headers);
64  struct string s;
65  init_string(&s);
66  int j;
67  struct curl_slist *header_list = NULL;
68  for (j = 0; j < headers->size; j++) {
69  char* header_key = lh_list_get(keys_of_headers, j);
70  char* header_value = (char*) lh_dict_get(headers, header_key);
71  uint16_t header_len = strlen(header_key) + strlen(header_value) + 3; // +3 because we will add ": " and the null terminating byte.
72  char* header_text = malloc(header_len * sizeof header_text);
73  header_text[0] = 0;
74  strcat(header_text, header_key);
75  strcat(header_text, ": ");
76  strcat(header_text, header_value);
77  header_list = curl_slist_append(header_list, header_text);
78  free(header_text);
79  }
80  lh_list_free(keys_of_headers);
81  curl_easy_setopt(curl, CURLOPT_URL, url);
82  if (method == PERFORM_POST) {
83  curl_easy_setopt(curl, CURLOPT_POST, 1);
84  if (request_body != NULL)
85  curl_easy_setopt(curl, CURLOPT_POSTFIELDS, request_body);
86 
87  }
88  if (method == PERFORM_PUT) {
89  curl_easy_setopt(curl, CURLOPT_CUSTOMREQUEST, "PUT");
90  if (request_body != NULL)
91  curl_easy_setopt(curl, CURLOPT_POSTFIELDS, request_body);
92  }
93  curl_easy_setopt(curl, CURLOPT_HTTPHEADER, header_list);
94  curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, writefunc);
95  curl_easy_setopt(curl, CURLOPT_WRITEDATA, &s);
96 
97  res = curl_easy_perform(curl);
98  if (res != CURLE_OK) {
99  char* message_template = "Request to url %s failed.";
100  char* message = malloc((strlen(url) + strlen(message_template) + 1) * sizeof message);
101  sprintf(message, message_template, url);
102  log_error(message);
103  free(message);
104  }
105  long http_response_code_long;
106  uint16_t* http_response_code = malloc(sizeof http_response_code);
107  curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &http_response_code_long);
108  *http_response_code = (uint16_t) http_response_code_long;
109  LH_HttpResponse* response = malloc(sizeof *response);
110  response->http_code = http_response_code;
111  response->response_body = s.ptr;
112  curl_slist_free_all(header_list);
113  curl_easy_cleanup(curl);
114  return response;
115  }
116  return NULL;
117 }
