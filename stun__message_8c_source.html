29 int stun_process_stun_message(uint8_t *bytes, uint16_t length, StunMessage *message) {
30  if (!stun_is_well_formed(bytes, length)) return 0;
31 
32  // initialize StunMessage
33  message->bytes = bytes;
34  message->length = length;
35  uint8_t message_type[2];
36  message_type[0] = bytes[0] >> 2;
37  message_type[1] = (bytes[1] & 0x0F) + ((bytes[1] & 0xE0) >> 1) + ((bytes[0] & 0x02) << 6);
38  message->method = byte_array_to_uint16_t(message_type);
39  message->api_key = NULL;
40  message_type[0] = bytes[0] & 0x01;
41  message_type[1] = bytes[1] & 0x10;
42  message->class = byte_array_to_uint16_t(message_type);
43  return 1;
44 }

46 int stun_get_attribute(const StunMessage *message, uint16_t attribute_code, StunAttribute *attribute) {
47  int position = STUN_MIN_MESS_LEN; // start reading bytes just after message header
48  while (position < message->length) {
49  uint16_t attr_type = byte_array_to_uint16_t(message->bytes + position);
50  uint16_t attr_length = byte_array_to_uint16_t(message->bytes + position + 2);
51  if (attr_type == attribute_code) {
52  attribute->attr_type = attr_type;
53  attribute->length = attr_length;
54  attribute->bytes = (message->bytes + position + 4);
55  return 1;
56  }
57  // jump to the next attribute
58  int shift = attr_length % 4;
59  if (shift != 0)
60  shift = 4 - shift;
61  // to take into account the padding to 32 bits boundary and the 4
62  // bytes of type length
63  position += attr_length + shift + 4;
64  }
65  // not found, return false
66  return 0;
67 }

69 int stun_is_well_formed(const uint8_t *bytes, uint16_t length) {
70 
71  // header of a STUN message is at least 20 bytes long
72  if (length < 20) {
73  log_warn("STUN message shorter than 20 bytes");
74  return STUN_FALSE;
75  }
76  // check two first bits are 00
77  if ((bytes[0] & 0xC0) != 0) {
78  log_warn("Two first bits in header are not zero");
79  return STUN_FALSE;
80  }
81 
82  uint32_t magic = MAGIC_COOKIE;
83 
84  uint8_t magicCookie[4];
85  uint32_to_byte_array(magic, magicCookie);
86  // check magic cookie
87 
88  if (!(bytes[4] == magicCookie[0] && bytes[5] == magicCookie[1]
89  && bytes[6] == magicCookie[2] && bytes[7] == magicCookie[3])) {
90  log_warn("Magic cookie is not correct");
91  return STUN_FALSE;
92  }
93  // check message length
94  uint16_t declared_length = byte_array_to_uint16_t(&bytes[2]);
95  if (declared_length != length - 20) {
96  log_warn("Incorrect message length");
97  return STUN_FALSE;
98  }
99 
100  return STUN_TRUE;
101 }

103 int stun_is_integrity_correct(const StunMessage *message, const char *api_key) {
104  const uint8_t *in = message->bytes;
105  uint16_t inlen = message->length - 24; // discard last 24 bytes (the message integrity)
106  uint16_t keylen = 36; // default because api keys are always uuids
107  uint8_t hmac_signature[20];
108  int fail = hmac_sha1(api_key, keylen, in, inlen, hmac_signature);
109 
110  // check for success of hmac calculation
111  if (fail) {
112  char error_message[70];
113  char trId[15];
114  encode_hex((message->bytes + 8), 12, trId);
115  snprintf(error_message, 70, "hmac_sha1 couldn't be determined for message with trId %s", trId);
116  log_error(error_message);
117  return STUN_FALSE;
118  }
119 
120  // check message integrity is correct
121  const uint8_t *declared_hmac_signature = (message->bytes + message->length - 20);
122  int hmac_incorrect = memcmp(declared_hmac_signature, hmac_signature, 20);
123  if (hmac_incorrect) {
124  char error_message[150];
125  char good_hmac[43], bad_hmac[43];
126  encode_hex(hmac_signature, 20, good_hmac);
127  encode_hex(declared_hmac_signature, 20, bad_hmac);
128  snprintf(error_message, 150, "hmac verification failed: should be %s but received %s", good_hmac, bad_hmac);
129  log_error(error_message);
130  return STUN_FALSE;
131  }
132  return STUN_TRUE;
133 }

135 uint16_t stun_get_error_code(const StunMessage *message, char *error_message_buffer, int buffer_len) {
136  if (message->class != CL_ERROR) {
137  log_warn("class of message was not CL_ERROR");
138  return 0;
139  }
140 
141  StunAttribute error_attribute;
142  if (!stun_get_attribute(message, ATTR_ERROR_CODE, &error_attribute)) {
143  log_warn("message did not contain ATTR_ERROR_CODE");
144  return 0;
145  }
146 
147  uint8_t hundreds = error_attribute.bytes[2];
148  uint8_t remainder = error_attribute.bytes[3];
149  uint16_t error_code = 100 * hundreds + remainder;
150 
151  uint16_t error_message_length = error_attribute.length - 4;
152  if (error_message_length > buffer_len - 1)
153  // allow room to add a null char at the end of the buffer
154  error_message_length = buffer_len - 1;
155  memcpy(error_message_buffer, error_attribute.bytes + 4, error_message_length);
156  error_message_buffer[error_message_length] = 0x00; //null terminate string
157  return error_code;
158 }

160 void stun_set_method_and_class(StunMessage *message, uint16_t method, uint16_t class) {
161  uint8_t type_bytes[2];
162  uint8_t method_bytes[2];
163  uint16_to_byte_array(method, method_bytes);
164  // encode method
165  type_bytes[1] = (0x0F & method_bytes[1]) + ((0x70 & method_bytes[1]) << 1);
166  type_bytes[0] = ((0x80 & method_bytes[1]) >> 6) + ((0x0F & method_bytes[0]) << 2);
167  // encode class
168  switch (class) {
169  case CL_REQUEST:
170  type_bytes[0] = type_bytes[0] & 0xFE; // set bit to 0
171  type_bytes[1] = type_bytes[1] & 0xEF; // set bit to 0
172  break;
173  case CL_SUCCESS:
174  type_bytes[0] = type_bytes[0] | 0x01; // set bit to 1
175  type_bytes[1] = type_bytes[1] & 0xEF; // set bit to 0
176  break;
177  case CL_ERROR:
178  type_bytes[0] = type_bytes[0] | 0x01; // set bit to 1
179  type_bytes[1] = type_bytes[1] | 0x10; // set bit to 1
180  break;
181  case CL_INDICATION:
182  type_bytes[0] = type_bytes[0] & 0xFE; // set bit to 0
183  type_bytes[1] = type_bytes[1] | 0x10; // set bit to 1
184  break;
185  }
186 
187  // set type bytes in STUN message
188  message->bytes[0] = type_bytes[0];
189  message->bytes[1] = type_bytes[1];
190 
191 }

203 StunMessage* stun_new_stun_message(uint16_t length, uint8_t set_tr_id) {
204  static uint8_t random_generator_initialized = STUN_FALSE;
205  if (!random_generator_initialized) {
206  srand(time(NULL));
207  random_generator_initialized = STUN_TRUE;
208  }
209 
210  StunMessage *stun_message = malloc(sizeof *stun_message);
211  stun_message->bytes = malloc(length * sizeof stun_message->bytes);
212  stun_message->length = length;
213  // set STUN message length to default minimum
214  uint16_to_byte_array(STUN_MIN_MESS_DECL_LEN, (stun_message->bytes + 2));
215  // set magic cookie
216  uint32_to_byte_array(MAGIC_COOKIE, (stun_message->bytes + 4));
217  // set random transaction id (12 random bytes)
218  if (set_tr_id) {
219  int j;
220  for (j = 0; j < 12; j = j + 2) {
221  uint8_t rand_bytes[2];
222  uint16_to_byte_array(rand() % 65536, rand_bytes);
223  stun_message->bytes[j + 8] = rand_bytes[0];
224  stun_message->bytes[j + 8 + 1] = rand_bytes[1];
225  }
226  }
227  return stun_message;
228 }

242 StunMessage* stun_set_message_integrity(StunMessage *message, char *api_key) {
243  uint16_t declared_length = byte_array_to_uint16_t(message->bytes + 2);
244  uint16_t length_after_mi_added = declared_length + 24;
245  uint16_to_byte_array(length_after_mi_added, message->bytes + 2);
246  // lower case api key
247  char *cursor = api_key;
248  for (;*cursor; ++cursor){
249  char a = *cursor;
250  a = tolower(a);
251  *cursor = a;
252  }
253  uint8_t message_integrity[STUN_ATTR_MESS_INTEGR_VALUE_LEN];
254  int fail = hmac_sha1(api_key, STUN_API_KEY_LEN, message->bytes, declared_length + STUN_MIN_MESS_LEN, message_integrity);
255  if (fail)
256  return NULL;
257 
258  // restore declared length to original value so that stun_add_attribute does not break the message
259  uint16_to_byte_array(declared_length, message->bytes + 2);
260 
261 
262  return stun_add_attribute(message, ATTR_MESSAGE_INTEGRITY, STUN_ATTR_MESS_INTEGR_VALUE_LEN, message_integrity);
263 }

265 StunMessage* stun_add_attribute(StunMessage *message, uint16_t attr_type, uint16_t value_length, uint8_t *value){
266  uint16_t declared_length = byte_array_to_uint16_t(message->bytes + 2);
267  uint8_t num_padding_bytes = value_length % 4;
268  if (num_padding_bytes != 0)
269  num_padding_bytes = 4 - num_padding_bytes;
270  // check if call to realloc is needed
271  uint16_t needed_length = declared_length + STUN_MIN_MESS_LEN + value_length + num_padding_bytes + 4;
272  if(needed_length > message->length){
273  // increase size of message
274  log_debug("Increasing size of message.");
275  message->bytes = realloc(message->bytes, needed_length * sizeof *message->bytes);
276  memset(message->bytes + message->length, needed_length - message->length, 0);
277  message->length = needed_length;
278  }
279  // set attribute length and type
280  uint8_t ptr_attr_start = declared_length + STUN_MIN_MESS_LEN;
281  uint8_t byte_array[2];
282  uint16_to_byte_array(attr_type, byte_array);
283  message->bytes[ptr_attr_start] = byte_array[0];
284  message->bytes[ptr_attr_start+1] = byte_array[1];
285  uint16_to_byte_array(value_length, byte_array);
286  message->bytes[ptr_attr_start+2] = byte_array[0];
287  message->bytes[ptr_attr_start+3] = byte_array[1];
288  // copy bytes
289  memcpy(message->bytes + ptr_attr_start + 4, value, value_length);
290  // set padding bytes
291  if (num_padding_bytes != 0){
292  int j;
293  for (j = 0; j < num_padding_bytes; j++){
294  message->bytes[ptr_attr_start+value_length + j + 4] = 0x00;
295  }
296  }
297  //update declared message length and byte array length
298  uint16_to_byte_array(needed_length - STUN_MIN_MESS_LEN, byte_array);
299  memcpy(message->bytes+2, byte_array, 2);
300  return message;
301 }
